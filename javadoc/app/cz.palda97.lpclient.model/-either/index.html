Either

sealed class Either<L, R>

Wrapper for two different object types. Intended to be used with the keyword when.

fun f(): Either<A, B>

val b: B = when(val res = f()) {

    is Either.Left -> return StatusCode.PARSING_ERROR

    is Either.Right -> res.value

}
